# 1. View running processes
ps
ps aux
ps -ef

# 2. Filter specific process
ps aux | grep firefox

# 3. Real-time process monitoring
top
htop  # (may need to install: sudo apt install htop)

# 4. Kill a process by PID
kill <PID>
kill -9 <PID>  # force kill

# 5. Kill a process by name
pkill firefox
pkill -f "python myscript.py"

# 6. Start a process in background
./script.sh &

# 7. List background jobs
jobs

# 8. Bring a background job to foreground
fg %1  # (1 is job number from 'jobs')

# 9. Send process to background
Ctrl + Z  # pause process
bg  # send it to background

# 10. Change process priority (nice value)
nice -n 10 command_name
renice -n 5 -p <PID>

# 11. Show process tree
pstree
pstree -p  # with PIDs

# 12. Display memory and CPU usage per process
ps aux --sort=-%mem | head
ps aux --sort=-%cpu | head

# 13. View open files by process
lsof -p <PID>

# 14. View which process is using a port
sudo lsof -i :8080
sudo netstat -tulpn | grep :8080

# 15. Monitor a specific process
watch -n 1 'ps -p <PID> -o %cpu,%mem,cmd'

# 16. Create a zombie process (for learning/testing)
(sleep 100 &); kill -STOP $!

# 17. Check process limits
ulimit -a
