# 1. Show IP address and network interfaces
ip addr show
ip a

# 2. Display active network connections and listening ports
netstat -tuln
ss -tuln

# 3. Show routing table
ip route
route -n

# 4. Ping a host to check connectivity
ping google.com
ping -c 4 google.com  # send only 4 packets

# 5. Test DNS resolution
nslookup google.com
dig google.com

# 6. Traceroute to check the path to a server
traceroute google.com

# 7. Download a file using curl or wget
curl -O https://example.com/file.txt
wget https://example.com/file.txt

# 8. Check open ports using nmap (you might need to install it)
sudo apt install nmap
nmap localhost
nmap -p 1-65535 localhost

# 9. Check hostname and domain name
hostname
hostname -f

# 10. Restart network manager (for troubleshooting)
sudo systemctl restart NetworkManager

# 11. Show current WiFi networks (for wireless)
nmcli device wifi list

# 12. Connect to a WiFi network using terminal (example)
nmcli device wifi connect "SSID_NAME" password "YOUR_PASSWORD"

# 13. Display network statistics
netstat -s

# 14. Display MAC address
ip link show
ifconfig  # also shows MAC (ifconfig may need to be installed)

# 15. Monitor live network traffic (requires iftop)
sudo apt install iftop
sudo iftop

# 16. Check public IP address
curl ifconfig.me
curl ipinfo.io/ip
