# 1. List files and directories
ls -l
ls -a
ls -lh

# 2. Change directory
cd /path/to/directory
cd ..
cd ~

# 3. Make a new directory
mkdir new_folder
mkdir -p parent_folder/child_folder

# 4. Remove files and directories
rm file.txt
rm -r folder_name
rm -rf folder_name  # force delete

# 5. Copy files and directories
cp file1.txt file2.txt
cp -r folder1 folder2

# 6. Move or rename files
mv oldname.txt newname.txt
mv file.txt /path/to/destination/

# 7. Display file contents
cat file.txt
less file.txt
more file.txt

# 8. Create an empty file
touch newfile.txt

# 9. Search text in files
grep "search_term" filename.txt
grep -r "search_term" /path/to/folder

# 10. Find files
find . -name "filename.txt"
find / -type f -name "*.log"

# 11. Display current directory
pwd

# 12. Display current date and time
date

# 13. Check disk usage
df -h
du -sh *

# 14. Show currently running processes
ps
ps aux
top

# 15. Change file permissions
chmod 755 script.sh
chmod +x script.sh

# 16. Change file ownership
sudo chown user:group file.txt

# 17. Download file from URL
wget https://example.com/file.txt
curl -O https://example.com/file.txt

# 18. Archive files
tar -cvf archive.tar file1 file2
tar -xvf archive.tar
tar -czvf archive.tar.gz foldername
tar -xzvf archive.tar.gz

# 19. Count words, lines, characters
wc file.txt
wc -l file.txt
wc -w file.txt
wc -c file.txt

# 20. Display last lines of a file
tail file.txt
tail -n 20 file.txt

# 21. Display first lines of a file
head file.txt
head -n 10 file.txt

# 22. Clear terminal
clear

# 23. Reboot or shutdown (requires sudo)
sudo reboot
sudo shutdown now
